\begin{exercise}

\outcome{Calculate limits using the limit laws.}
\outcome{Calculate limits of the form 0/0.}

\tag{limits}

Find
\[
\lim_{x\to5^{+}}\left(\frac{x^2-4x-5}{\left|x-5\right|}\right)
\begin{prompt}
= \answer{6}
\end{prompt}
\]

\begin{hint}
Multiply by the conjugate.
\end{hint}
