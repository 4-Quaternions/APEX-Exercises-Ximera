\begin{exercise}
  %% An "outcome" should be what skills/knowledge a student demonstrates
  %% by solving this problem.
  %%
  %% Imagine an outcome is prefaced with the words "The learner can..."
  %%
  %% The outcomes should come from the list at 
  %%   https://github.com/mooculus/mooculus/blob/master/outcomesList.tex
  %% or at the outcomes (not the topics) listed at
  %%  https://github.com/mooculus/mooculus/blob/master/textbook/README.md
  \outcome{Evaluate limits of continuous functions.}
  \outcome{Use limit laws.}

  %% Tags should be one- or two-word phrases that give an objective
  %% description of the problem.
  \tag{limit} 
  \tag{quadratic} 
  \tag{continuous}

  % Here is what the problem goes
  Compute:
  \[
  \lim_{x\to 1} \left( x^2+3x-5 \right)
  \begin{prompt}
    % The answer should be written as a TeX expression; 
    = \answer{-1}
  \end{prompt}
  \]
  
  \begin{hint}
    Broadly speaking, we want to know which number $x^2+3x-5$ is near
    when $x$ is near $1$.
  \end{hint}
  
  \begin{hint}
    By applying the limit law that the limit of a sum is the sum of the limits, we learn that
    \[
    \lim_{x\to 1} \left( x^2+3x-5 \right)  
    = \lim_{x\to 1} \left( x^2 \right) +
    \lim_{x\to 1} \left( 3x \right) +
    \lim_{x\to 1} \left( -5 \right).
    \]
  \end{hint}
  
  \begin{hint}
    Additional limit laws imply that
    \[
    \lim_{x\to 1} \left( x^2+3x-5 \right)  
    = \left( \lim_{x\to 1} \left( x \right) \right)^2 +
    3 \cdot \lim_{x\to 1} \left( x \right) +
    \lim_{x\to 1} \left( -5 \right).
    \]
  \end{hint}
  
  \begin{hint}
    But the $\lim_{x\to 1} \left( x \right) = 1$ and $\lim_{x\to 1} \left( -5 \right) = -5$.
  \end{hint}
  
  % The last hint should always "give away" the answer.
  \begin{hint}
    Therefore
    \[
    \lim_{x\to 1} \left( x^2+3x-5 \right)  
    = \left( \lim_{x\to 1} 1 \right)^2 +
    3 \cdot 1 + 
    -5 = 1 + 3 - 5 = -1.
    \]
  \end{hint}
  
\end{exercise}
